typedef int Item;
typedef struct no {
   Item item;
   struct no *prox;
} *Lista;


// -----------------------------------------------------------------------------
// Exemplo 2
// -----------------------------------------------------------------------------

Lista no(Item x, Lista p) {
   Lista n = malloc(sizeof(struct no));
   n->item = x;
   n->prox = p;
   return n;
}


// -----------------------------------------------------------------------------
// Exemplo 4
// -----------------------------------------------------------------------------

void exibe(Lista L) {
   while( L != NULL ) {
      printf("%d\n",L->item);
      L = L->prox;
   }
}

Lista aleatoria(int n, int m) {
   Lista L = NULL;
   while( n>0 ) {
      L = no(rand()%m, L);
      n--;
   }
   return L;
}

void anexa(Lista *A, Lista B) {
   if( !B ) return;
   while( *A )
      A = &(*A)->prox;
   *A = B;
}

void destroi(Lista *L) {
   while( *L ) {
      Lista n = *L;
      *L = n->prox;
      free(n);
   }
}
